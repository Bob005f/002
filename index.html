Befehle:

Klonen

cd ./(Ordnername) --> In ordner/Verzeichniss gehen

cd.. (Aus dem Ordner/akutellem verzeichniss gehen)

git clone (SSH-key der webstite) --> Klont die Dateien

Hochladen/Pushen

git commit -m "name"

git add *

git push --> Lädt die Dateien hoch

Branch

git branch --> zeigt aktuelle Branch

git branch "name" --> Erstellt Branch

git checkout branch "name" --> Wechselt zur Branch "name"

git branch megrge "name" --> aktuelle Branch wird mit Branch"name" gemergt/zusammengefügt

git branch -d "name" --> löscht Branch name
